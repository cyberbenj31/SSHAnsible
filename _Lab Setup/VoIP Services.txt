  service rivanaa out-bound
  destination-pattern #$3AT#69
  session target ipv4:10.#$3AT#.100.8
  incoming called-number #$3AT#69
  dtmf-relay h245-alphanumeric
  codec g711ulaw
  no vad

====================
Session Initiation Protocol (SIP)
Configuration

!@cucm
conf t
 voice service voip
  allow-connections h323 to sip
  allow-connections sip to h323
  allow-connections sip to sip
  supplementary-service h450.12
 sip
   bind control source-interface fa0/0
   bind media source-interface fa0/0
   registrar server expires max 600 min 60
!
 voice register global
  mode cme
  source-address 10.#$3AT#.100.8 port 5060
  max-dn 12
  max-pool 12
  authenticate register
  create profile sync
 voice register dn 1
   number #$3AT#34
   allow watch
   name #$3AT#34
 voice register dn 2
   number #$3AT#43
   allow watch
   name #$3AT#43
!
  voice register pool 1
    id mac #M0B1LEMAC1#
    number 1 dn 1
    dtmf-relay sip-notify
    username #$3AT#34 password #$3AT#34
    codec g711ulaw
!
  voice register pool 2
    id mac #M0B1LEMAC2#
    number 1 dn 2
    dtmf-relay sip-notify
    username #$3AT#43 password #$3AT#43
    codec g711ulaw
	end
